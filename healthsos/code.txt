#include <stdio.h>
int main()
{int k=0,i,s,j,r;
 scanf("%d",&r);
 for(i=r;i>0;i--)
 {
     for(s=0;s<r-i;s++)
     printf(" ");
    for(j=1;j<=2*i-1;j++)
    printf("%d",i);
    
    printf("\n");
 }
 for(i=2;i<=r;i++,k=0)
 {
     for(s=0;s<=r-i-1;s++)
     printf(" ");
     while(k!=2*i-1)
     {
         printf("%d",i);
         k++;
     }
     printf("\n");
 }
 
    

    return 0;
}
